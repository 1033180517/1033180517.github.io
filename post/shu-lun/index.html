费马小定理

如果一个数p是质数，那么有
a ^ (p - 1) ≡ 1 (mod p)
a ^ p ≡ a (mod p)

拓展欧几里德算法

存在整数对(x,y),使得 gcd(a, b) = ax + by，求解x,y
时间复杂度O(logn)

//递归代码  
int exgcd(int a, int b, int&x, int&y) {  
    if (b == 0) {  
        x = 1;  
        y = 0;  
        return a;  
    }  
    int r = exgcd(b, a%b, y, x);  
    int t = x;  
    y = y - a/b*t;  
    return r;  
}
// 非递归算法  
int exgcd(int m, int n, int &x, int &y) {  
    int x1, y1, x0, y0;  
    x0 = 1; y0 = 0;  
    x1 = 0; y1 = 1;  
    x = 0; y = 1;  
    int r = m%n;  
    int q = (m-r)/n;  
    while(r) {  
        x = x0 - q*x1;  
        y = y0 - q*y1;  
        x0 = x1; y0 = y1;  
        x1 = x; y1 = y;  
        m = n; n = r; r = m%n;  
        q = (m-r)/n;  
    }  
    return n;
}
