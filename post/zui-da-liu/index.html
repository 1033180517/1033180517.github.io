最大流

2020-07-08 算法

Ford-Fulkerson （dfs写法）

int n,m,s,t;
int x,y,z;
int vis[10002];

struct edge{
    int to,wei,re;
};
vector<edge>Map[10002];

void add_edge(int a,int b,int w){
    Map[a].push_back({b,w,Map[b].size()});
    Map[b].push_back({a,0,Map[a].size()-1});
}

int dfs(int x,int num){
    if(x==t)return num;
    vis[x]=1;
    for(int i=0;i<Map[x].size();i++){
        edge &e=Map[x][i];
        if(!vis[e.to]&&e.wei>0){
            int t=dfs(e.to,min(num,e.wei));
            if(t>0){
                e.wei-=t;
                Map[e.to][e.re].wei+=t;
                return t;
            }
        }
    }
    return 0;
}

int max_flow(int x){
    int flow=0;
    for(;;){
        memset(vis,0,sizeof(vis));
        int f=dfs(x,inf);
        flow+=f;
        if(f==0)return flow;
    }    
}

int main(){
    ios::sync_with_stdio(0);
    cin>>n>>m>>s>>t;
    for(int i=0;i<m;i++){
        cin>>x>>y>>z;
        add_edge(x,y,z);
    }
    cout<<max_flow(s)<<endl;
}
