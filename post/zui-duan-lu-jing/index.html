最短路径

2020-07-08 算法

1.Floyd O(n^3)

//暴力松弛，优化后，Map[i][j]表示结点i到结点j的最短距离

int n，Map[1005][1005];
int main(){
cin>>n;
for(int i=1;i<=n;i++){
for(int j=1;j<=n;j++){
scanf(“%d”,&Map[i][j]);
}
}
for(int k=1;k<=n;k++){
for(int i=1;i<=n;i++){
for(int j=1;j<=n;j++){
if(Map[i][j]>Map[i][k]+Map[k][j]){
Map[i][j]=Map[i][k]+Map[k][j];
}
}
}
}
}
2.朴素Dijkstra O(n^2)

//单源最短路，优化后，dis[i]表示结点s到结点i的最短距离

int n,m,s,s1,s2,s3,Map[10002][10002];
int dis[10002],vis[10002];
int main(){
cin>>n>>m>>s;
for(int i=1;i<=n;i++){
for(int j=1;j<=n;j++){
Map[i][j]=inf;
}
Map[i][i]=0;
}
while(m–){
scanf(“%d%d%d”,&s1,&s2,&s3);
if(s3<Map[s1][s2])Map[s1][s2]=s3;
if(s3<Map[s2][s1])Map[s2][s1]=s3;
}
for(int i=1;i<=n;i++){
dis[i]=Map[s][i];
}
memset(vis,0,sizeof(vis));
vis[s]=1;
for(int i=1;i<=n;i++){
int u=inf,p=-1;
for(int j=1;j<=n;j++){
if(!vis[j]&&u>dis[j]){
u=dis[j];
p=j;
}
}
vis[p]=1;
for(int j=1;j<=n;j++){
if(!vis[j]&&dis[j]>dis[p]+Map[p][j]){
dis[j]=dis[p]+Map[p][j];
}
}
}
}
