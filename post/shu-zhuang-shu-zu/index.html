int tree[500003]; //维护sub
int tree1[500003];
int tree2[500003]; //维护 sub2
int sub2[500002]; //表示sub[i]*i的前缀和

//表示x的二进制表示从右向左第一个1换成十进制为多少

inline int lowbit(int x){
return x&(-x);
}

inline int range_query(int l,int r){
int ans1=0,ans2=0;
for(int i=r;i>0;i-=lowbit(i)){
ans1+=tree[i];
}
for(int i=l-1;i>0;i-=lowbit(i)){
ans2+=tree[i];
}
return ans1-ans2;
}

tree1[j]+=sub[i];
}
}
for(int i=1;i<=n;i++){
for(int j=i;j<=n;j+=lowbit(j)){
tree2[j]+=sub[i]*i;
}
}
}

//区间修改

void range_add(int l,int r,int d){
for(int i=l;i<=n;i+=lowbit(i)){
tree1[i]+=d;
}
for(int i=r+1;i<=n;i+=lowbit(i)){
tree1[i]-=d;
}
for(int i=l;i<=n;i+=lowbit(i)){
tree2[i]+=d*l;
}

for(int i=r+1;i<=n;i+=lowbit(i)){
    tree2[i]-=d*r-d;
}

}

//区间查询

int range_query(int l,int r){
int ans1=0,ans2=0,s=0;

for(int i=r;i>0;i-=lowbit(i)){
    s+=tree1[i];
}
ans1+=s*(r+1);
s=0;
for(int i=r;i>0;i-=lowbit(i)){
    s+=tree2[i];
}
ans1-=s;
s=0;
for(int i=l-1;i>0;i-=lowbit(i)){
    s+=tree1[i];
}
ans2+=s*l;
s=0;
for(int i=l-1;i>0;i-=lowbit(i)){
    s+=tree2[i];
}
ans2-=s;
return ans1-ans2;

}
